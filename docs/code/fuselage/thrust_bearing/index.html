In experimenting with OpenSCAD I started off trying to build a wire thrust bearing. That effort was partially successful, bu tin the end, I decided to go with as bent aluminum beaing. This took a bit of work, but it was not so bad. Basically, is is a a simple piece of rectangular stock with rounded pieces on the bearing ends.

Here is the data file needed:

10thb_width = 1/16;
11thb_front_height = thb_height/3;

16thrust_bearing_color = METAL_Aluminium;

The rectangular part of he bearing was created using a 2D shape that was then extruded to form the basic part, Two cylinderical sections wee added to cap off this part.

Here is the 2D file, created as a library file:

 1//######################################
 2// prop_bearing_2D.scad
 3// (c) Roie R. Black
 4//**************************************
 5use <MMlib/circular_arc_strip.scad>
 6$fn=100;
 7
 8module prop_bearing_2D(
 9  length = 0.4,         // prop to rear
10  height = 0.15,         // top to prop-shaft
11  front_height = 3/64,  // nose height
12  top_length = 0.2,     // top flat length
13  thickness = 1/64      // material thickness
14) {
15  bend_radius = 2 * thickness;
16
17  // geometry calculations
18  d = height - bend_radius;
19  ky = d - front_height;
20  kx = length - 2 * bend_radius - top_length;
21  b = sqrt(kx*kx + ky*ky);
22  alpha = atan2(kx,ky);
23  echo(b,d,kx,ky,alpha,thickness);
24
25  // locate bend centers
26  p1 = [2 * thickness, front_height];
27  p2 = p1 + [kx,ky];
28  p3 = [length - 2 * thickness,d];
29
30  // front flat
31  square([thickness,front_height]);
32
33  // first bend at p1
34  translate(p1) circular_arc_strip(bend_radius,180-alpha,180,thickness);
35
36  // sloped front flat
37  translate(p1 + [-bend_radius * cos(alpha), bend_radius*sin(alpha)])
38    rotate(-alpha)
39      square([thickness, b]);
40
41  // second bend at p2
42  translate(p2) circular_arc_strip(bend_radius, 90,180-alpha,thickness);
43
44  // top flat
45  translate([p2.x,p2.y + thickness]) square([top_length,thickness]);
46
47  // third bend at p3
48  translate(p3) circular_arc_strip(bend_radius, 0,90,thickness);
49  // segment 5 - top
50  // back flat
51  translate([length - thickness,0])
52    square([thickness,d]);
53}
54
55//---------------------------------------
56// debug display
57
58prop_bearing_2D();

And the code that generates the part:

 1//#####################################
 2// thrust-bearing.scad
 3// (c) 2021 - Roie R. Black
 4//=====================================
 5include <thrust_bearing_data.scad>
 6
 7use <MMlib/prop_bearing_2D.scad>
 8$fn=100;
 9
10module thrust_bearing() {
11  color(thrust_bearing_color)
12    difference() {
13      union() {
14        translate([0,thb_width/2,0])
15          rotate([90,0,0])
16            linear_extrude(thb_width)
17              prop_bearing_2D(
18                thb_length,
19                thb_height,
20                thb_front_height,
21                thb_top_length,
22                thb_thickness
23              );
24        // front lower cap
25        rotate([0,90,0])cylinder(r=thb_width/2,h=thb_thickness);
26        // rear cap
27        translate([thb_length-thb_thickness,0,0])
28          rotate([0,90,0])
29            cylinder(r=thb_width/2, h=thb_thickness);
30      }
31      translate([-0.1,0,0])
32        rotate([0,90,0])
33          cylinder(r=thb_wire_size/2,thb_length + 0.2);
34    }
35}
36
37thrust_bearing();
